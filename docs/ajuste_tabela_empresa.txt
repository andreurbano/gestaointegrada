/*------------------------------------------------------------------------------
SELECT com os totais de manifestos por empresa.
*/------------------------------------------------------------------------------
 SELECT   e.ID_Empresa,
          e.RazaoSocial,
          COUNT(m.GerID_Empresa) AS Total
 FROM     manifesto m RIGHT JOIN empresa e ON m.GerID_Empresa = e.ID_Empresa
 GROUP BY e.ID_Empresa
 ORDER BY e.ID_Empresa;
